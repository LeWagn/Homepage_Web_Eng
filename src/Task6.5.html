Deep Copy

const deepCopy = (struct) => Array.isArray(struct) ? struct.map(deepCopy) :
typeof struct === 'object' ? Object.fromEntries(Object.entries(struct).map(([k,v])=> [k,deepCopy(v)])) :
struct;

const originalArray = [1,2,3];
const originalObject = {a:1,b:2};
const originalNestedObject = {a:[1,2,3],b:2};
console.assert(deepCopy(originalArray) !== originalArray);
console.assert(deepCopy(originalObject) !== originalObject);
console.assert(deepCopy(originalNestedObject) !== originalNestedObject);

console.log(deepCopy(originalArray));
console.log(deepCopy(originalObject));
console.log(deepCopy(originalNestedObject));

Returns
[ 1, 2, 3 ]
{ a: 1, b: 2 }
{ a: [ 1, 2, 3 ], b: 2 }
