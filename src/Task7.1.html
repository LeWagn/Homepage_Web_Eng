function identity_function(x) {

};

7.1.2

function addf(x) {
return function(y) {
return x + y;
}
}

7.1.3

function applyf(binaryFunc) {
return function(x) {
return function(y) {
return binaryFunc(x, y);
};
};
}

7.1.4

function curry(binaryFunc, x) {
return function(y) {
return binaryFunc(x, y);
};
}

7.1.6

function methodize(binaryFunc) {
return function(y) {
return binaryFunc(this, y);
}
}

7.1.7

function demethodize(unaryMethod) {
return function(x, y) {
return unaryMethod.call(x, y);
}
}

7.1.8

function twice(binaryFunction) {
return function(x) {
return binaryFunction(x, x);
}
}

var double = twice(function(x, y) { return x + y });
console.log(double(11)); // 22

var square = twice(function(x, y) { return x * y });
console.log(square(11)); // 121

function composeu(func1, func2) {

}

7.1.10

function composeb(a, b) {
return function(x, y, z) {
return a(b(x, y), z);
}
}

7.1.11

function once(fn) {
let called = false;
return function() {
if (!called) {
called = true;
return fn.apply(this, arguments);
}
throw new Error("Function can only be called once.");
}
}

7.1.12

function counterf(start) {
let count = start;
function inc() {
return ++count;
}
function dec() {
return --count;
}
return { inc, dec };
}

7.1.13

function revocable(func) {
return {
invoke: function(x) {
if(!this.revoked) {
return func(x);
} else {
throw "Error: Funktion wurde zurückgenommen und kann nicht mehr aufgerufen werden.";
}
},
revoke: function() {
this.revoked = true;
}
}
}
